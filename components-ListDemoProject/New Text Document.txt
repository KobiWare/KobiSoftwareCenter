package components;

import java.awt.*;
import java.awt.event.*;
import javax.swing.*;
import java.beans.*;
import java.util.Random;

public class ZipperProgress extends JPanel
                             implements ActionListener, 
                                        PropertyChangeListener {

    private JProgressBar progressBar;
    private JTextArea taskOutput;
    private Task task;
    private final int SPEED = 1000;
    private static Item item;

    class Task extends SwingWorker<Void, Void> {
        /*
         * Main task. Executed in background thread.
         */
        int progress = 0;
        
        @Override
        public Void doInBackground() {
            Random random = new Random();
            //Initialize progress property.
            
            
            Copy a = new Copy(item);
            
            
            new Thread(() -> {
               //Do whatever
               try {
                  
                  String msg = a.copyTo("C:\\Users\\nathan.mills\\Desktop\\test\\" + item.toString());
                  System.out.println("DONE!!");
                  //progress.setToDone();
                  if(msg.equals("")) {
                     JOptionPane.showMessageDialog(null, "Done!");  
                  } else {
                     JOptionPane.showMessageDialog(null, msg, null, JOptionPane.ERROR_MESSAGE);
                  }
               } catch(Exception IOException) {
                  JOptionPane.showMessageDialog(null, "Error: " + IOException);  
               }
            }).start();
            
            setProgress(0);
            while (progress < 100) {
                try {
                     Thread.sleep(SPEED / (100 - progress));
                } catch (InterruptedException ignore) {}
                if(progress < 99) {
                     progress += 1;
                }
                setProgress(Math.min(progress, 100));
            }
            return null;
        }

        /*
         * Executed in event dispatching thread
         */
        @Override
        public void done() {
            Toolkit.getDefaultToolkit().beep();
            taskOutput.append("Done!\n");
        }
    }

    public ZipperProgress(Item item) {
        super(new BorderLayout());
        
        this.item = item;

        //Create the demo's UI.

        progressBar = new JProgressBar(0, 100);
        progressBar.setValue(0);
        progressBar.setStringPainted(true);

        taskOutput = new JTextArea(5, 20);
        taskOutput.setMargin(new Insets(5,5,5,5));
        taskOutput.setEditable(false);

        JPanel panel = new JPanel();
        panel.add(progressBar);

        add(panel, BorderLayout.PAGE_START);
        add(new JScrollPane(taskOutput), BorderLayout.CENTER);
        setBorder(BorderFactory.createEmptyBorder(20, 20, 20, 20));
        
        task = new Task();
        task.addPropertyChangeListener(this);
        task.execute();

    }
    
    // TODO: Remove
    public void actionPerformed(ActionEvent evt){}

    /**
     * Invoked when task's progress property changes.
     */
    public void propertyChange(PropertyChangeEvent evt) {
        if ("progress" == evt.getPropertyName()) {
            int progress = (Integer) evt.getNewValue();
            progressBar.setValue(progress);
            taskOutput.append(String.format(
                    "Completed %d%% of task.\n", task.getProgress()));
        } 
    }


    /**
     * Create the GUI and show it. As with all GUI code, this must run
     * on the event-dispatching thread.
     */
    public static void createAndShowGUI() {
        //Create and set up the window.
        System.out.println("Made UI");
        JFrame frame = new JFrame("Installing " + item.toString());
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

        //Create and set up the content pane.
        JComponent newContentPane = new ZipperProgress(item);
        newContentPane.setOpaque(true); //content panes must be opaque
        frame.setContentPane(newContentPane);

        //Display the window.
        frame.pack();
        frame.setVisible(true);
    }
    
    static boolean done = false;
    
    public static void setToDone() {
      done = true;
    }
    
    public static void progressBar() {
         javax.swing.SwingUtilities.invokeLater(new Runnable() {
            public void run() {
               // TODO: return from progress bar method, not just this silly excuse of a fix
               if(done)
                  return;
               System.out.println("progressBar");
               System.out.println(done);
               createAndShowGUI();
            }
        });
    }
}
